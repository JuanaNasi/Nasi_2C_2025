153void MostrarDistancia(void *pvParameter)
154{
155 while (1)
156 {
157 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
158
159 if (control_medicion && !control_hold)
160 {
161 LcdItsE0803Write(distancia_medida);
162
163 // UartSendString: manda un string al puerto que paso
164 /* UartItoa: convierte un entero sin signo de 32 bits a una
165 cadena de caracteres en una base especificada y devuelve un puntero a
166 la cadena resultante. La cadena resultante se almacena en un búfer
167 estático buf de tamaño 32.*/
168
169 if(control_unidad)
170 {
171 UartSendString(UART_PC,"Distancia medida: ");
172 UartSendString(UART_PC, (const char*)UartItoa(distancia_medida, 10));
173 UartSendString(UART_PC," ");
174 UartSendString(UART_PC,"cm");
175 UartSendString(UART_PC,"\r\n"); //esto es el fin de linea abajo
176 }
177
178 else if(!control_unidad)
179 {
180 UartSendString(UART_PC,"Distancia medida: ");
181 UartSendString(UART_PC, (const char*)UartItoa(distancia_medida*constante_pulgadas, 10));
182 UartSendString(UART_PC," ");
183 UartSendString(UART_PC,"pulgadas");
184 UartSendString(UART_PC,"\r\n"); //esto es el fin de linea abajo
185 }
186 }
187
188 else if (!control_medicion)
189 {
190 LcdItsE0803Off();
191 }
192 // vTaskDelay(MEDICION_PERIODO / portTICK_PERIOD_MS);
193 }

248void ManejoPorTeclado()
249{
250 uint8_t tecla_ingresada;
251
252 UartReadByte(UART_PC, &tecla_ingresada);//va a leer la tecla que apreto en el teclado y me va a modificar mi variable
253
254 if (tecla_ingresada == 'O')
255 {
256 FuncionTecla1();
257 }
258
259 else if (tecla_ingresada == 'H')
260 {
261 FuncionTecla2();
262 }
263
264 else if (tecla_ingresada == 'I')
265 {
266 control_unidad = !control_unidad;
267 }
268
269 else if (tecla_ingresada == 'M')
270 {
271 UartSendString(UART_PC,"Distancia maxima: ");
272 UartSendString(UART_PC, (const char*)UartItoa(distancia_auxiliar, 10));
273 UartSendString(UART_PC," ");
274 UartSendString(UART_PC,"cm");
275 UartSendString(UART_PC,"\r\n"); //esto es el fin de linea abajo
276 }
277
278 else
279 {
280 printf("Tecla ingresada incorrecta.\n");
281 }
282}

287void app_main(void)
288{
289 /*Actividad 3 - Proyecto: Medidor de distancia por ultrasonido c/interrupciones y puerto serie
290
291 Modifique la actividad del punto 2 agregando ahora el puerto serie.
292 Envíe los datos de las mediciones para poder observarlos en un terminal en la PC,
293 siguiendo el siguiente formato:
294
295 3 dígitos ascii + 1 carácter espacio + dos caracteres para la unidad (cm) + cambio de línea “ \r\n”
296 Además debe ser posible controlar la EDU-ESP de la siguiente manera:
297 Con las teclas “O” y “H”, replicar la funcionalidad de las teclas 1 y 2 de la EDU-ESP
298 */
299
300 LedsInit();
301 HcSr04Init(GPIO_3, GPIO_2);
302 LcdItsE0803Init();
303 SwitchesInit();
304
305 /*Configuro e inicializo el puerto serie*/
306 serial_config_t mi_serial;
307 mi_serial.port = UART_PC;
308 mi_serial.baud_rate = 115200; // tasa de bits por segundo
309 mi_serial.func_p = ManejoPorTeclado; //esta funcion va a hacer las interrupciones cuando haya apretado una tecla
310
311 /*Configuro e inicializo la UART*/
312 UartInit(&mi_serial);
313
314 /*inicializo el timer*/
315 timer_config_t timer_distancia = {
316 .timer = TIMER_A,
317 .period = MEDICION_PERIODO,
318 .func_p = FuncTimerA,
319 .param_p = NULL};
320 TimerInit(&timer_distancia);
321
322 SwitchActivInt(SWITCH_1, &LeerSwitch, NULL); //interrupción
323 SwitchActivInt(SWITCH_2, &LeerSwitch, NULL);
324
325 xTaskCreate(&MedirDistancia, "Medir distancia", 2048, NULL, 5, &medir_distancia_handle); //tarea
326 xTaskCreate(&MostrarDistancia, "Mostrar distancia", 2048, NULL, 5, &mostrar_distancia_handle);
327
328 TimerStart(timer_distancia.timer);
329}
